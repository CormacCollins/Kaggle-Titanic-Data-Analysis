def resample_age_values(df):
    mean = np.mean(df['Age'])
    std = np.std(df['Age'])

    rand_age = np.random.randint(mean - std, mean + std, size = df['Age'].isnull().sum())
    age_slice = df["Age"].copy()
    age_slice[np.isnan(age_slice)] = rand_age
    df["Age"] = age_slice
    df["Age"] = df["Age"].astype(int)

I'm going to run these models with a few settings:

  • An Early stopping monitor with an arbitrary level of 3. That is, if there are no greater improvements in the 'loss function' after 3 tries the network will finish
  • I am using the optimizer 'sgd'
  • Loss funciton 'categorical_crossentropy'
  • An Epoch limit of 30
  • And a validation split of 0.3, that is the model will use 70% of the data on training, and test it on 30%
  • All Activation functions will arbitrarily be 'relu'

Many of these functions are because my understanding is still basic, I will look to learn how to tweak and optimize these in the future.

def create_model(layer_num, density_arr):
    early_stopping_monitor = EarlyStopping(patience=3) 

    # ----------- Model 1 ----------------------

    # Set up the model: model
    model = Sequential()

    # Add the first layer
    model.add(Dense(density_arr[0], activation='relu', input_shape=(n_cols,)))

    if layer_num > 1:
        for i in range(1, layer_num):
            # Add the second layer
           # print(density_arr)
            #print(i)
            model.add(Dense(density_arr[i], activation='relu'))

    # Add the output layer
    model.add(Dense(2, activation='softmax'))

    # Compile the model
    model.compile(optimizer='sgd', 
                  loss='categorical_crossentropy', 
                  metrics=['accuracy'])
        # Fit the model
